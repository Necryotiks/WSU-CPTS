template <class T>
class BSTNode
{
public:
	BSTNode();
	~BSTNode();
	T getData();
	BSTNode<T> *& getLeft();
	BSTNode<T> *& getRight();
	bool setData(T &newData);
private:
	T data;
	BSTNode<T> * LPTR;
	BSTNode<T> * RPTR;
};

template<class T>
BSTNode<T>::BSTNode()
{
	data = '\0';
	LPTR = nullptr;
	RPTR = nullptr;
}

template<class T>
BSTNode<T>::~BSTNode()
{
}

template<class T>
T BSTNode<T>::getData()
{
	return T(data);
}

template<class T>
BSTNode<T> *& BSTNode<T>::getLeft()
{
	return LPTR;
}

template<class T>
BSTNode<T> *& BSTNode<T>::getRight()
{
	return RPTR;
}

template<class T>
bool BSTNode<T>::setData(T &newData)
{
	data = newData;
	if(data != newData)
	{
		return false;
	}
	return true;
}
