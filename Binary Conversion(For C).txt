void binaryconvertnbits(int input,int n)
{
	int j = 0; //counter
	int bit = 0;
	int number[n] = { 0 };//store bit number here
	for (j = 0; j < n; j++)
	{
		bit = input & 1;//mask all but least sig.bit
		number[n - j - 1] = bit;
		input = input >> 1;//right shift
	}
	for (j = 0; j < n; j++)
	{
		printf("%d", number[j]);
	}
}
