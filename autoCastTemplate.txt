template<typename T, typename U>
T autoCast(U castTarget)
{
	return static_cast<T>(castTarget);
}

//Casts user input.
//Useful for preventing users from breaking your defined program type inputs(i.e do while menu,switch);
//Will return default value on data input mismatch.
//<int> is default seeting for use with numbered menus.
